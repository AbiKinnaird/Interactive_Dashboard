def RandomForest(dataframe, num_samples):
    """
    This function implements the Random Forest model

    :param dataframe: Credit card fraud data
    :param num_samples: number of samples to train the model on
    :return: outputs a dictionary of the random forest model and the trained X and Y data

    """
    #Preprocessing into the test and training sets
    df = dataframe.sample(n=num_samples, random_state=42)
    y = df['Class']
    X = df.drop('Class', axis = 1)
    X_train, X_test, Y_train, Y_test = train_test_split(X, y, random_state=0)

    #Apply SMOTE to even the dataset
    smote = SMOTE(random_state=99)
    X_train_resampled, Y_train_resampled = smote.fit_resample(X_train, Y_train)

    #Train the model
    rf = RandomForestClassifier(n_estimators=100, max_depth = 5,
                                random_state=99)
    rf.fit(X_train_resampled, Y_train_resampled)

    return {'df':df,'rf':rf,'X_train':X_train,'Y_train':Y_train, 'X_resampled':X_train_resampled, 'Y_resampled':Y_train_resampled, 'X_test':X_test,'Y_test':Y_test}

    image_path = 'baobab.png'
    ax.get_figure().set_size_inches(6,4.8)
    ax.get_figure().savefig(image_path, format='png', dpi=300, transparent=False, bbox_inches='tight', pad_inches=0.1)

    with Image.open(image_path) as img:

        cropped_img = img.crop((280, 130, img.width - 200, img.height - 90))
        cropped_image_path = 'baobab_cropped.png'
        cropped_img.save(cropped_image_path, format='png', dpi=(300, 300))

    with open(cropped_image_path, 'rb') as image_file:
        image = base64.b64encode(image_file.read())
        image = "data:image/png;base64, " + image.decode()

    return image
